#include <stdio.h>

int main()
{
    int a, b, c;

    printf("Enter three numbers: ");
    scanf("%d %d %d", &a, &b, &c);

    if (a == b && b == c)
    {
        printf("All three numbers are equal: %d", a);
    }
    else if (a >= b && a >= c)
    {
        printf("A is the biggest number: %d", a);
    }
    else if (b >= a && b >= c)
    {
        printf("B is the biggest number: %d", b);
    }
    else
    {
        printf("C is the biggest number: %d", c);
    }

    return 0;
}
